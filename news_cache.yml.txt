name: Refresh News Cache

on:
  schedule:
    - cron: '7 * * * *'   # once an hour (offset a few min past the hour)
  workflow_dispatch:        # lets you trigger it manually from the Actions tab

permissions:
  contents: write   # needed so the workflow can commit news_cache.json back

jobs:
  refresh-news-cache:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: '3.12'

      - name: Install dependencies
        run: pip install requests

      - name: Fetch news calendar
        run: python3 fetch_news_cache.py

      - name: Commit updated cache
        run: |
          git config user.name "news-cache-bot"
          git config user.email "news-cache-bot@users.noreply.github.com"
          git add news_cache.json
          git diff --staged --quiet || git commit -m "Refresh news_cache.json [skip ci]"
          git push
